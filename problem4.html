#include <iostream>
using namespace std;

void func1(int &limit) { 
    cin >> limit;
}

void func3(int num) {  
    cout << num << endl;
}

void func2(int limit) { 
    for (int i = 1; i <= limit; i++) {
        if (i % 4 == 0 && i % 7 == 0) {
            func3(i);
        }
    }
}

int main() {
    int n;
    func1(n);
    func2(n);

    return 0;
}
