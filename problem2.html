#include <iostream>
using namespace std;

void func1(string &inputStr) {
    cout << "Enter a string: ";
    cin >> inputStr;
}

string func2(string originalStr) {
    string tempStr = "";
    int length = originalStr.size();
    for (int i = length - 1; i >= 0; i--) {
        tempStr += originalStr[i];
    }
    return tempStr;
}

void func3(string firstStr, string secondStr) {
    if (firstStr == secondStr) {
        cout << "Yes" << endl;
    } else {
        cout << "No" << endl;
    }
}

int main() {
    string userInput;
    func1(userInput);

    string reversedStr = func2(userInput);
    func3(userInput, reversedStr);

    return 0;
}
