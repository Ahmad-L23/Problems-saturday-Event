#include <iostream>
using namespace std;

void funcName1(int &value) {
    cout << "Enter a number: ";
    cin >> value;
}

void funcName2(int limit, int &result) {
    result = 0;
    for (int i = 1; i <= limit; i++) {
        if (i % 2 != 0) {
            result += i;
        }
    }
}

void funcName3(int limit, int result) {
    cout << result << endl;
}

int main() {
    int inputValue, finalResult;
    funcName1(inputValue);
    funcName2(inputValue, finalResult);
    funcName3(inputValue, finalResult);

    return 0;
}
